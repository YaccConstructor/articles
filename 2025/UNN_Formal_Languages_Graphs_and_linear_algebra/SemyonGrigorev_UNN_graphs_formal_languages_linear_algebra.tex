  \frametitle{Статичекий анализ кода}
  \begin{itemize}
    \item Контекстно-свободные языки (CFL-r, Context-Free Language Reachability)
    \begin{itemize}
      \item Томас Репс --- основоположник
      \begin{itemize}
        \item Одна из основополагающих статей: \href{https://dl.acm.org/doi/10.1145/199448.199462}{Precise interprocedural dataflow analysis via graph reachability}, 1995
      \end{itemize}
      \item Относительно свежий обзор области: \href{https://dl.acm.org/doi/abs/10.1145/3583660.3583664}{CFL/Dyck Reachability: An Algorithmic Perspective}, 2023
      \item Активно разрабатываются инструменты: \href{https://ieeexplore.ieee.org/abstract/document/10623254}{Pearl: A Multi-Derivation Approach to Efficient CFL-Reachability Solving}, 2024
      \item Есть открытые теоретические проблемы. Например: \href{https://dl.acm.org/doi/abs/10.1145/3498702}{Subcubic certificates for CFL reachability}, 2023
    \end{itemize} 
    \item За границами контекстно-свободных языков
    \begin{itemize} 
      \item \href{https://dl.acm.org/doi/10.1145/3093333.3009848}{Context-sensitive data-dependence analysis via linear conjunctive language reachability}, 2017
      \item \href{https://dl.acm.org/doi/10.1145/3704854}{Program Analysis via Multiple Context Free Language Reachability}, 2025
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Графовые БД}
  \begin{itemize}
    \item Регулярные языки (RPQ, Regular Path Queries)
    \begin{itemize}
      \item Часть ISO-стандартов языков запросов к рафам
      \begin{itemize}
        \item Расширение SQL для работы с граф-структурированными данными: \href{https://www.iso.org/standard/79473.html}{ISO/IEC 9075-16:2023 Information technology --- Database languages SQLPart 16: Property Graph Queries (SQL/PGQ)}
        \item Язык запросов к графам: \href{https://www.iso.org/standard/76120.html}{ISO/IEC 39075:2024 Information technology --- Database languages --- GQL}
      \end{itemize}
      \item Частично поддерживается в разных реализациях \href{https://docs.cambridgesemantics.com/anzograph/v2.5/userdoc/cyref-patterns.htm#patterns-relationships}{Cypher}\footnote{Разделы <<Patterns for Relationships>>, <<Variable-length Pattern Matching>>}
      \item Одна из классических работ: \href{https://dl.acm.org/doi/abs/10.1145/2274576.2274585}{Regular path queries on graphs with data}, 2012
      \item Даже теория ещё не вся изучена: \href{https://lmcs.episciences.org/12597}{Fine-Grained Complexity of Regular Path Queries}, 2023
    \end{itemize} 
    \item Контекстно-свободные языки (CFPQ, Context-Free Path Queries)
    \begin{itemize} 
      \item Михалис Яннакакис --- основоположник
      \begin{itemize}
        \item Одна из основополагающих статей: \href{https://dl.acm.org/doi/abs/10.1145/298514.298576}{Graph-theoretic methods in database theory}, 1990
      \end{itemize}
      \item \href{https://www.degruyterbrill.com/document/doi/10.1515/jib-2008-100/html}{Subgraph Queries by Context-Free Grammars}, 2008
      \item \href{https://www.semanticscholar.org/paper/Understanding-Data-Science-Lifecycle-Provenance-via-Miao-Deshpande/d8b098a729ae32a11182eb075b0d48e61f7613e3}{Understanding Data Science Lifecycle Provenance via Graph Segmentation and Summarization}, 2018
    \end{itemize}
  \end{itemize}
\end{frame}
